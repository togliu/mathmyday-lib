package io.github.ltennstedt.finnmath.linear.matrix

/**
 * Immutable implementation of a matrix entry
 *
 * @param E type of the element
 * @property rowIndex row index
 * @property columnIndex column index
 * @property element element
 * @constructor Constructs a [MatrixEntry]
 * @throws IllegalArgumentException if [rowIndex] < 1
 * @throws IllegalArgumentException if [columnIndex] < 1
 * @author Lars Tennstedt
 * @since 0.0.1
 */
public data class MatrixEntry<E : Number>(
    val rowIndex: Int,
    val columnIndex: Int,
    val element: E
) : Comparable<MatrixEntry<E>> {
    init {
        require(rowIndex > 0) { "expected rowIndex > 0 but rowIndex = $rowIndex" }
        require(columnIndex > 0) { "expected columnIndex > 0 but columnIndex = $columnIndex" }
    }

    override fun compareTo(other: MatrixEntry<E>): Int = COMPARATOR.compare(this, other)

    public companion object {
        /**
         * [Comparator]
         *
         * @since 0.0.1
         */
        @JvmField
        public val COMPARATOR: Comparator<MatrixEntry<*>> =
            compareBy(MatrixEntry<*>::rowIndex, MatrixEntry<*>::columnIndex)
    }
}
